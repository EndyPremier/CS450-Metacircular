(define f
  (lambda (a b)
    (display (locally-defined? a))
    (display (locally-defined? b))
    (locally-make-unbound! a)
    (locally-make-unbound! b)
    (display (locally-defined? a))
    (display (locally-defined? b))
    (display (defined? c))
    (display (defined? b))
    (make-unbound! c)
    (make-unbound! b)
    (display (defined? c))
    (display (defined? b))
  )
)
